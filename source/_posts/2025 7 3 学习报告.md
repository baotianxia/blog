## [洛谷P3403 跳楼机](https://www.luogu.com.cn/problem/P3403)

同余最短路

我们选`xyz`中的最小值作为`mod`，然后围绕这三个数进行同余最短路，最后再正常计数就行了，基本上是板子

```cpp
#include <iostream> 
#include <cstring>
#include <algorithm>
#include <queue>
#define ll long long
using namespace std;
ll dis[100005];
ll l[3];
queue<ll>q;
int main(){
 ll h;
 cin>>h;
 for(int i=0;i<3;i++){
  cin>>l[i];
 }
 sort(l,l+3);
 ll mod=l[0];
 memset(dis,0x7f,sizeof(dis));
 dis[0]=0;
 q.push(0);
 while(!q.empty()){
  ll u=q.front();
  q.pop();
  for(int k=1;k<3;k++) {
      ll v=(u+l[k])%mod;
      if(dis[u]+l[k]<dis[v]) {
          dis[v]=dis[u]+l[k];
          q.push(v);
      }
  }
 }
 ll ans=0;
 for(int i=0;i<mod;i++){
  if(dis[i]<=h){
   ans+=(h-1-dis[i])/mod+1;
//   if(dis[i]==0) ans--;// 排除地面（0层） 
  }
  
 }
 cout<<ans;
 return 0;
}
```
